This tool provides an estimate only. Final VA decisions depend on the medical record, exams, and adjudication.
Quick Claim Inputs
async function runAnalyze() { try { const token = localStorage.getItem("token"); if (!token) { window.location.href = "/login"; return; } const condition = document.getElementById("serviceContext")?.value || "Claim"; const in_service_event = document.getElementById("inServiceEvent")?.checked || false; const current_diagnosis = document.getElementById("currentDiagnosis")?.checked || false; const nexus_letter = document.getElementById("nexusLetter")?.checked || false; const severity = document.getElementById("severity")?.value || "moderate"; const res = await fetch("/analyze", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": "Bearer " + token }, body: JSON.stringify({ condition, in_service_event, current_diagnosis, nexus_letter, severity }) }); const data = await res.json(); if (!res.ok || !data.success) { if (res.status === 401) { localStorage.removeItem("token"); window.location.href = "/login"; return; } alert(data.error || "Analyze failed"); return; } const r = data.result || {}; const el = document.getElementById("resultCard"); el.classList.remove("hidden"); el.innerHTML = `

Claim Result

Condition: ${r.condition || "Unknown"}
Service Connection: ${r.service_connection || "Unknown"}
Estimated Rating: ${r.estimated_rating || "Unknown"}
Confidence: ${r.confidence || "Unknown"}
Why:
${r.why || ""}
Biggest Lever:
${r.biggest_lever || ""}
What’s Missing:
    ${(r.missing || []).map(item => `
  • ${item}
    • `).join("")}
Next Steps:
    ${(r.next_steps || []).map(step => `
  • ${step}
    • `).join("")}
C&P Advice:
${r.cp_advice || ""}
Save this claim
Get this result and next steps emailed to you.
`; } catch (err) { console.log(err); alert("Something broke"); } }
